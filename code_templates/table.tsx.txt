import React, { useState, FC } from 'react';
import { TceTable } from '@tencent/tce-component';
import { getListFields, generateAddOn } from './config';
import { useTableData } from './hooks';
import './index.css';

const Table: FC<> = function({}) {

  const [page, setPage] = useState(1);
  const [pageSize, setPageSize] = useState(20);

  // table 数据获取
  const {
    records,
    totalRecords,
    loading,
  } = useTableData(page, pageSize)

  // 定义插件
  const addons = generateAddOn();

  // 定义字段
  const columns = getListFields();

  function onPagingChange({ pageIndex, pageSize }) {
    setPage(pageIndex);
    setPageSize(pageSize);
    return true;
  }

  return (
    <div className="">
      <TceTable
        loading={loading}
        records={records}
        addons={addons}
        bordered={true}
        columns={columns}
        rowDisabled={record => record['loading'] === true}
        pageIndex={page}
        recordCount={totalRecords}
        pageSize={pageSize}
        onPagingChange={onPagingChange}
      />
    </div>
  );
};

export default Table;
