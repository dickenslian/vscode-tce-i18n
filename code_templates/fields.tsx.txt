import React from 'react';
import { t, Trans } from '@tea/app/i18n';

export function getListFields( ) {
  // 列表参数
  let columns = [
    {
      header: t('部署状态'),
      key: 'status',
      width: 100,
      render: data => {
        const status = data.status || '';
        return status.toLowerCase() === 'unknown' ? '-' : status;
      }
    }
  ];

  return columns;
}

export default getListFields;
